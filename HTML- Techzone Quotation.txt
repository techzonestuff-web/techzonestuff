<script>
    document.getElementById('q-date').valueAsDate = new Date();
    document.getElementById('q-no').innerText = 'TZ-' + Math.floor(100000 + Math.random() * 900000);

    const categories = [
        "Processor (CPU)", "Motherboard", "Memory (RAM)", "Solid State Drive (SSD)", 
        "Hard Disk Drive (HDD)", "Graphics Card (GPU)", "Power Supply (PSU)", 
        "Cabinet / Chassis", "CPU Cooler / Liquid Cooler", "Monitor", 
        "Laptop", "Desktop PC (Branded)", "All-in-One PC", 
        "Keyboard & Mouse", "UPS / Inverter", "Networking (Router/Switch)", 
        "Printer / Scanner", "Cables / Adapters / Converters", 
        "Speakers / Headsets", "Webcam / Mic", "External Storage",
        "Operating System (Windows)", "Antivirus / Software", 
        "Service / Labor / Assembly", "Annual Maintenance (AMC)", "Other (Type in Spec)"
    ];

    function addRow() {
        const tbody = document.getElementById('table-body');
        const row = document.createElement('tr');
        let options = categories.map(cat => `<option value="${cat}">${cat}</option>`).join('');
        
        row.innerHTML = `
            <td><select onchange="handleOtherSelect(this)">${options}</select></td>
            <td><input type="text" placeholder="Brand/Model/Spec"></td>
            <td><input type="number" value="1" min="1" oninput="calculateTotals()" class="qty"></td>
            <td><input type="number" value="0" min="0" oninput="calculateTotals()" class="price"></td>
            <td class="row-total" style="font-weight:bold;">0.00</td>
            <td class="no-print" style="text-align:center;"><button class="btn-remove" onclick="removeRow(this)">×</button></td>
        `;
        tbody.appendChild(row);
        calculateTotals();
    }

    function handleOtherSelect(select) {
        if(select.value === "Other (Type in Spec)") {
            const row = select.closest('tr');
            row.querySelector('input[type="text"]').placeholder = "Type custom category here...";
            row.querySelector('input[type="text"]').focus();
        }
    }

    function removeRow(btn) { btn.closest('tr').remove(); calculateTotals(); }

    function calculateTotals() {
        const rows = document.querySelectorAll('#table-body tr');
        let subtotal = 0;
        rows.forEach(row => {
            const qty = parseFloat(row.querySelector('.qty').value) || 0;
            const price = parseFloat(row.querySelector('.price').value) || 0;
            const total = qty * price;
            row.querySelector('.row-total').innerText = total.toFixed(2);
            subtotal += total;
        });
        const gstRate = parseFloat(document.getElementById('gst-rate').value);
        const discount = parseFloat(document.getElementById('discount').value) || 0;
        const gstAmount = (subtotal * gstRate) / 100;
        const grandTotal = (subtotal + gstAmount) - discount;

        document.getElementById('subtotal').innerText = subtotal.toLocaleString('en-IN', {minimumFractionDigits: 2});
        document.getElementById('gst-amount').innerText = gstAmount.toLocaleString('en-IN', {minimumFractionDigits: 2});
        document.getElementById('grand-total').innerText = '₹ ' + grandTotal.toLocaleString('en-IN', {minimumFractionDigits: 2});
        document.getElementById('amount-in-words').innerText = numberToIndianWords(Math.round(grandTotal)) + " Rupees Only";
    }

    function numberToIndianWords(num) {
        const a = ['', 'One ', 'Two ', 'Three ', 'Four ', 'Five ', 'Six ', 'Seven ', 'Eight ', 'Nine ', 'Ten ', 'Eleven ', 'Twelve ', 'Thirteen ', 'Fourteen ', 'Fifteen ', 'Sixteen ', 'Seventeen ', 'Eighteen ', 'Nineteen '];
        const b = ['', '', 'Twenty', 'Thirty', 'Forty', 'Fifty', 'Sixty', 'Seventy', 'Eighty', 'Ninety'];
        const format = (n) => (n < 20) ? a[n] : b[Math.floor(n / 10)] + (n % 10 !== 0 ? ' ' + a[n % 10] : '');
        if (num === 0) return 'Zero';
        let str = '';
        if (num >= 10000000) { str += format(Math.floor(num / 10000000)) + 'Crore '; num %= 10000000; }
        if (num >= 100000) { str += format(Math.floor(num / 100000)) + 'Lakh '; num %= 100000; }
        if (num >= 1000) { str += format(Math.floor(num / 1000)) + 'Thousand '; num %= 1000; }
        if (num >= 100) { str += format(Math.floor(num / 100)) + 'Hundred '; num %= 100; }
        if (num > 0) str += format(num);
        return str.trim();
    }

    window.onload = addRow;
