<script>
    // Tab Management
    function showPage(pageId, btn) {
        document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
        document.getElementById(pageId).classList.add('active');
        btn.classList.add('active');
    }

    // GST Tool Logic
    function calculateGST() {
        const mode = document.getElementById("mode").value;
        const amount = parseFloat(document.getElementById("amount").value) || 0;
        const gstRate = parseFloat(document.getElementById("gst").value) || 0;
        const res = document.getElementById("gstResult");
        
        if (mode === "add") {
            let gst = amount * (gstRate / 100);
            res.innerHTML = `Base: ₹${amount.toLocaleString()} | GST: ₹${gst.toLocaleString()} | <b>Total: ₹${(amount + gst).toLocaleString()}</b>`;
        } else {
            let base = amount / (1 + (gstRate / 100));
            res.innerHTML = `Incl: ₹${amount.toLocaleString()} | Base: ₹${base.toLocaleString()} | <b>Tax Part: ₹${(amount - base).toLocaleString()}</b>`;
        }
    }

    // Estimate Tool Logic
    const componentOptions = [
        "Processor (CPU)", "Motherboard", "Memory (RAM)", "Solid State Drive (SSD)",  
        "Hard Disk Drive (HDD)", "Graphics Card (GPU)", "Power Supply (PSU)", 
        "Cabinet / Chassis", "CPU Cooler / Liquid Cooler", "Monitor", 
        "Laptop", "Desktop PC (Branded)", "All-in-One PC", 
        "Keyboard & Mouse", "UPS / Inverter", "Networking (Router/Switch)", 
        "Printer / Scanner", "Cables / Adapters / Converters", 
        "Speakers / Headsets", "Webcam / Mic", "External Storage",
        "Operating System (Windows)", "Antivirus / Software", 
        "Service / Labor / Assembly", "Annual Maintenance (AMC)", "Other (Type in Spec)"
    ];

    function addItem() {
        const container = document.getElementById("itemsContainer");
        const row = document.createElement("div");
        row.className = "item-row";
        let opts = componentOptions.map(opt => `<option value="${opt}">${opt}</option>`).join('');
        row.innerHTML = `
            <div><label>Category</label><select class="category">${opts}</select></div>
            <div><label>Details</label><input type="text" class="spec" placeholder="Model/Brand"></div>
            <div><label>Qty</label><input type="number" class="qty" value="1"></div>
            <div><label>Rate (₹)</label><input type="number" class="rate" value="0"></div>
            <button class="btn btn-danger remove-btn" onclick="this.parentElement.remove()">✕</button>
        `;
        container.appendChild(row);
    }

    function generateEstimate() {
        const rows = document.querySelectorAll(".item-row");
        const gstRate = parseFloat(document.getElementById("quoteGstRate").value) || 0;
        let subtotal = 0;

        let html = `<table><thead><tr><th>Category</th><th>Description</th><th>Qty</th><th>Unit Rate</th><th>Total</th></tr></thead><tbody>`;
        
        rows.forEach(row => {
            const cat = row.querySelector(".category").value;
            const spec = row.querySelector(".spec").value || "---";
            const qty = parseFloat(row.querySelector(".qty").value) || 0;
            const rate = parseFloat(row.querySelector(".rate").value) || 0;
            const itemTotal = qty * rate;
            subtotal += itemTotal;
            html += `<tr><td>${cat}</td><td>${spec}</td><td>${qty}</td><td>₹${rate.toLocaleString()}</td><td>₹${itemTotal.toLocaleString()}</td></tr>`;
        });

        const gstAmount = subtotal * (gstRate / 100);
        const grandTotal = subtotal + gstAmount;

        html += `</tbody></table>`;
        html += `
            <div class="summary-box">
                <div class="summary-row"><span>Sub-Total:</span><span>₹${subtotal.toLocaleString(undefined, {minimumFractionDigits: 2})}</span></div>
                <div class="summary-row"><span>GST (${gstRate}%):</span><span>₹${gstAmount.toLocaleString(undefined, {minimumFractionDigits: 2})}</span></div>
                <div class="grand-total"><span>Grand Total:</span><span>₹${grandTotal.toLocaleString(undefined, {minimumFractionDigits: 2})}</span></div>
            </div>
        `;
        
        document.getElementById("estimateOutput").innerHTML = html;
    }

    // Default row on startup
    addItem();
</script>
